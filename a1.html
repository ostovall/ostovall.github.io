Resistor Calculation

R = (5V − Vf) / I. With Vf ≈ 2.0–2.2V and I = 10 mA, R ≈ 280–300 Ω. I used 330 Ω for safety and part availability. P ≈ 0.033 W.

  // Declare an integer variable to store the pin number for the red LED
int redLED = 2;

// Declare an integer variable to store the pin number for the yellow LED
int yellowLED = 3;

// Declare an integer variable to store the pin number for the green LED
int greenLED = 4;

// The setup() function runs once when the Arduino powers on or resets
void setup() {
  // Set the red LED pin to OUTPUT mode so the Arduino can send voltage to it
  pinMode(redLED, OUTPUT);

  // Set the yellow LED pin to OUTPUT mode so the Arduino can send voltage to it
  pinMode(yellowLED, OUTPUT);

  // Set the green LED pin to OUTPUT mode so the Arduino can send voltage to it
  pinMode(greenLED, OUTPUT);
}

// The loop() function runs repeatedly after setup() finishes
void loop() {
  // Turn the red LED ON by sending 5 volts to its pin
  digitalWrite(redLED, HIGH);

  // Turn the yellow LED OFF to make sure only red is lit
  digitalWrite(yellowLED, LOW);

  // Turn the green LED OFF to make sure only red is lit
  digitalWrite(greenLED, LOW);

  // Wait for one second (1000 milliseconds) before changing LEDs
  delay(1000);

  // Turn the red LED OFF before switching to yellow
  digitalWrite(redLED, LOW);

  // Turn the yellow LED ON by sending 5 volts to its pin
  digitalWrite(yellowLED, HIGH);

  // Keep the green LED OFF while yellow is on
  digitalWrite(greenLED, LOW);

  // Wait for one second before switching to the next LED
  delay(1000);

  // Make sure the red LED is OFF
  digitalWrite(redLED, LOW);

  // Make sure the yellow LED is OFF
  digitalWrite(yellowLED, LOW);

  // Turn the green LED ON by sending 5 volts to its pin
  digitalWrite(greenLED, HIGH);

  // Wait for one second before repeating the pattern
  delay(1000);
}
